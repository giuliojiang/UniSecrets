SELECT
  `post`.`college` AS college,
  `likes` AS likes,
  post.`text` AS posttext,
  `dislikes` AS dislikes,
  `comment`.text AS commenttext,
  `user`.`nickname` AS commentnickname
FROM
  `post`
LEFT JOIN
  `comment` ON post.postid = `comment`.commentid
LEFT JOIN
  `user` ON `comment`.email = `user`.`email`
WHERE
  `post`.`college` = "Imperial College London" OR `public` = 1
